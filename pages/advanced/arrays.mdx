## Arrays in Javascript

```js filename="arrays.js" {3} copy
// ======================================================
 
 
console.log('Access Individual Elements Of An Array');
let colors = ['red', 'green', 'blue']
console.log('colors: colors[0]: ', colors[0]) // Output: red
console.log('colors: colors[1]: ', colors[1]) // Output: green
console.log('colors: colors[2]: ', colors[2]) // Output: blue
console.log('')

// ======================================================
 
 
console.log('Adding items to an array')
colors.push('purple')
console.log('colors: ', colors) // ["red","green","blue","purple"]
console.log()


console.log('')

// ======================================================
 
console.log('Removing items from the array')
let removedColor = colors.pop()
console.log('colors: ', colors) // Output ["red","green","blue"]
console.log('removedColor:  ', removedColor); //Output: "purple"
console.log('')
 
// ======================================================

```
